дҪҶиҝҷйўҳж•°жҚ®иҢғеӣҙдёә5 * 10^4 еҰӮжһңarr[i]зӣёзӯүзҡ„иҫ№жңүmжқЎ йӮЈд№Ҳз”ұдәҺжҳҜдёҖдёӘж— еҗ‘еӣҫ йӮЈд№Ҳиҝҷж ·зҡ„йҒҚеҺҶзҡ„е°Ҷдјҡn^2 ж¬Ў жүҖд»ҘиҝҷдёҖжӯҘжҳҜдјҳеҢ–зҡ„е…ій”®
дјҳеҢ–зҡ„жҖқи·ҜжҳҜ жҳҜдҪҝз”Ёbfs жұӮжңҖзҹӯи·Ҝзҡ„и·Ҝеҫ„ bfsжұӮжңҖзҹӯи·ҜеҸӘжңүз¬¬дёҖж¬Ўжӣҙж–°ж—¶жұӮеҮәзҡ„еҖјжҳҜжңҖе°ҸеҖј жүҖд»ҘеҪ“дҪ дёҖж¬Ўжү©еұ•еҖјеқҮдёәvзҡ„зӮ№еҗҺ зӣҙжҺҘеҲ йҷӨvеҚіеҸҜ дёӢдёҖж¬ЎдёҚеҝ…еҶҚйҒҚеҺҶ еҚіеҜ№дәҺзӣёеҗҢеҖјеҫ—жүҖжңүзӮ№еҸӘиө°дёҖйҒҚ

class Solution:
    def minJumps(self, arr: List[int]) -> int:
        # е…Ҳи®°еҪ•жүҖжңүдёҺеҖјзӣёеҗҢзӮ№зҡ„дёӢж Ү
        h = defaultdict(list)
        for i, v in enumerate(arr):
            h[v].append(i)

        # йҳҹеҲ—еӯҳ дёӢж Ү
        q = deque()
        q.append(0)
        n = len(arr)
        st = n * [inf]
        st[0] = 0
        while q:
            t = q.popleft()
            v = arr[t]

            for ne in t-1,t+1:
                if 0 <= ne < n and st[ne] > st[t] + 1:            
                    st[ne] = st[t] + 1
                    q.append(ne)
                if ne == n-1:
                    return st[ne]

            if not h[v]:
                continue

            for ne in h[v][::-1]:
                if st[ne] > st[t] + 1:
                    st[ne] = st[t] + 1
                    q.append(ne)
                if ne == n-1:
                    return st[ne]
                # ж•ҙдёӘдјҳеҢ–жөҒзЁӢ жңҖе…ій”®зҡ„дёҖжӯҘ жҜҸиҝҷдёҖжӯҘе°ұдјҡи¶…ж—¶
                h[v] = None

        return st[n-1]

``
